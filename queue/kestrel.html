客户端

Ruby

sudo apt install -y libsasl2-dev   # ubuntu
sudo gem install kestrel-client

require 'kestrel'
q = Kestrel::Client.new('localhost:22133')
q.set('a_queue', 'foo')
q.get('a_queue') # => 'foo'
q.pook('a_queue') # => 'foo'
